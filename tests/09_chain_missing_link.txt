# Chaining With link Missing or Negation

# Test 1: Simple two-step chain
A => !B
!B => C
# Expected: C = true

# Test 2: Three-step chain
D => E
E => !F
F => G
# Expected: G = false

# Test 3: Branching chain
H => I
!I => J
I => K
# Expected: J = false, K = true

# Test 4: Diamond chain
L => M
L => !N
M + N => O
# Expected: M = true, N = false, O = false
