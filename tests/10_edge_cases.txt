# Edge Cases Test - Circular Dependencies and Negation Loops

# Test 1: Simple loop without forcing rule
# A => B, B => A (both should remain false without external trigger)
A => B
B => A
# Expected: A = false, B = false

# Test 2: Simple 2-node loop with forcing rule (true)
# C => D, D => C (should both become true)
C => D
D => C
# Expected: C = true, D = true

# Test 3: Loop with forcing rule (false via negation)
# F => !G, G => H, H => G (G and H should remain false)
F => !G
!G => !H
!H => !G
# Expected: G = false, H = false

# I => J, J => K, K => I (all three should be true if I set)

# Expected: I = true, J = true, K = true
